\begin{frame}[shrink=8]{Dataset Design: Four Initial-Condition Families}
\begin{columns}[T]
\column{0.58\textwidth}
\textbf{Shared velocity initialization}
\[
  \theta_i \sim \mathcal{U}(0,2\pi),
  \qquad
  \mathbf{v}_i(0)=v_0(\cos\theta_i,\sin\theta_i)^\top .
\]

\textbf{Spatial families}
\begin{align*}
  \mathbf{x}_i(0) &\sim \mathcal{U}(\Omega)
  && \text{uniform}, \\
  \mathbf{x}_i(0) &\sim \mathcal{N}(\boldsymbol{\mu},\sigma^2 I_2)
  && \text{single Gaussian}, \\
  \theta_i &\sim \mathcal{U}(0,2\pi), \quad
  r_i \sim \mathcal{N}(R_{\mathrm{ring}},\sigma_r^2)
  && \text{ring}, \\
  p(\mathbf{x})
  &=
  \tfrac12\mathcal{N}(\boldsymbol{\mu}_1,\sigma^2 I_2)
  +
  \tfrac12\mathcal{N}(\boldsymbol{\mu}_2,\sigma^2 I_2)
  && \text{two-cluster}.
\end{align*}

\smallskip
These ICs stress homogeneous, localised, rotationally organised, and bimodal density structure under the same simulator.

\begin{frame}[shrink=8]{What Is Held Fixed Across the Benchmark}
\small
\begin{columns}[T]
\column{0.54\textwidth}
\[
  \Delta t_{\mathrm{ROM}}
  =
  \Delta t \times \text{subsample}
  =
  0.04 \times 3
  =
  0.12\ \mathrm{s},
  \qquad
  m=19.
\]

\begin{itemize}
  \item periodic domain, $N=300$, $64\times 64$ KDE grid
  \item bandwidth $\sigma=5.0$ grid cells
  \item shift alignment enabled for every ROM run
  \item $\sqrt{\cdot}$ density transform used for both MVAR and LSTM
  \item mass rescaling applied after lifting
\end{itemize}

\begin{frame}[shrink=8]{Identification Quality Is Measured in Weak Form}
\small
\textbf{Weak-form fit}
\[
  R^2_{\mathrm{wf}}
  =
  1-
  \frac{\|\mathbf{b}-\mathbf{G}\mathbf{w}^*\|_2^2}
       {\|\mathbf{b}\|_2^2}
\]

\textbf{Term importance}
\[
  \Pi_m
  =
  |w_m^*|\frac{\|\mathbf{G}_{:,m}\|_2}{\|\mathbf{b}\|_2},
  \qquad
  \tilde\Pi_m
  =
  \Pi_m / \max_j \Pi_j .
\]

\textbf{Operational strong-identification criteria from the thesis}
\begin{itemize}
  \item continuity backbone coefficient in $[-1.01,-0.95]$
  \item physically plausible retained signs
  \item backbone terms stable across six support-scale triplets
  \item $R^2_{\mathrm{wf},\rho}>0.99$ and momentum quality high enough to support the mechanism pattern
\end{itemize}

\[
  \text{strong identification}
  \not\Rightarrow
  \text{adequate closure}
\]
\end{frame}

\begin{frame}[shrink=10]{Pure Vicsek: WSINDy Validates against Kinetic Theory}
\begin{columns}[T]
\column{0.50\textwidth}
\textbf{Mean-field kinetic theory predicts:}
\[
  \partial_t\rho + v_0\nabla\!\cdot\mathbf{p} = 0
\]
\[
  (p_x)_t = \underbrace{\beta\,\partial_x\rho}_{\text{pressure}}
           + \underbrace{\delta\,\nabla\!\cdot(p_x\mathbf{p})}_{\text{self-advection}}
\]
{\small $W\equiv 0$ (no Morse), $\beta=v_0^2/\bar\rho$, $\delta=v_0$}
\column{0.50\textwidth}
\textbf{WSINDy recovers (no tuning):}
\[
  \rho_t = -\nabla\!\cdot\mathbf{p}
\]
\[
  (p_x)_t = \beta\,\partial_x\rho + \delta\,\nabla\!\cdot(p_x\mathbf{p})
\]
{\small $R^2_{\mathrm{wf},\rho}=1.000,\enspace R^2_{\mathrm{wf},p}=0.82$}\\[0.3em]
{\small Normalised balance: $\Pi_{\delta}=1.000,\;\Pi_{\beta}=0.963$}\\[0.1em]
{\small No spurious Morse term selected ($\Pi_{\Phi}<\lambda$).}
\end{columns}
\centering\medskip\large\textbf{Correct leading-order mechanism recovery from data alone.}
\end{frame}

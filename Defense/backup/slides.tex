% ------ Backup 7: MSTLS Algorithm ------
\begin{frame}[noframenumbering]{MSTLS Algorithm: Sequential Thresholding}

\begin{enumerate}
  \item \textbf{Initialize:} OLS fit $\hat{\mathbf{w}} = \Gmat^\dagger\bvec$
  \item \textbf{Compute} dominant-balance ratios:
    $\Pi_m = \frac{|w_m|\,\|\mathbf{g}_m\|_2}{\|\bvec\|_2}$
  \item \textbf{Threshold:} set $w_m{=}0$ if $\Pi_m < \alpha\,\max_j\Pi_j$ \quad ($\alpha{=}0.1$)
  \item \textbf{Re-fit} on active support: $\hat{\mathbf{w}}_{\mathcal{S}} = \Gmat_{\mathcal{S}}^\dagger\bvec$
  \item \textbf{Iterate} steps 2--4 until support stabilizes
  \item \textbf{Cross-scale:} repeat for 6 support-scale triplets, retain intersection
\end{enumerate}

\bigskip
Dominant-balance criterion (Callaham et al.\ 2021) normalizes by column norm, preventing large-scale terms from dominating.
\end{frame}

% ------ Q&A: Closure Problem Definition ------
\begin{frame}[noframenumbering,shrink=6]{From Agents to PDEs: The Closure Problem}
\small
From microscopic trajectories the thesis observes $\mathbf{q}=(\rho,p_x,p_y,\Phi)$
with $\Phi=W*\rho$.
The density must satisfy a \textbf{conservation law}:
\[ \partial_t\rho + \nabla\!\cdot\mathbf{j}=0. \]
The \textbf{closure problem}: identify an effective flux law
$\mathbf{j}_{\boldsymbol\theta}[\rho,\mathbf{p},\Phi]$
such that the above matches data.

\medskip Scalar density alone does not close: alignment survives in $\mathbf{p}$; Morse interactions survive in $\Phi$.
\end{frame}

% ------ New Backup 1: Mass Preservation ------
\begin{frame}[noframenumbering]{Mass Preservation under Linear POD}
\centering
\normalsize
\textbf{Theorem (linear POD preserves mass)}\\[0.8em]
If $\mathbf{x}_k = U_m\mathbf{y}_k + \boldsymbol{\mu}$ and $\mathbf{1}^\top U_m = \mathbf{0}$ (mean-centred basis), then
\[
  \mathbf{1}^\top\mathbf{x}_k = \mathbf{1}^\top\boldsymbol{\mu} = M_0 \quad\forall\,k,
\]
so the lifted reconstruction has the same total mass as the training mean.

\medskip
\begin{itemize}
  \item In practice, the $\sqrt{\cdot}$ transform breaks linearity: reconstruction requires mass rescaling $\mathcal{S}_{M_0}$.
  \item Mass overshoot in any lifted snapshot is $\le 0.3\%$ in all tested regimes.
\end{itemize}
\end{frame}

% ------ New Backup 2: Weak-Form Design Matrix Assembly ------
\begin{frame}[noframenumbering,shrink=4]{Weak-Form Design Matrix Assembly}
\small
\[
  b_k = -\!\!\sum_{i,j,n}\!\!\partial_t\psi_k(x_i,y_j,t_n)\cdot u_{ij}^n\;\Delta x\,\Delta y\,\Delta t,
\]
\[
  G_{k,m} = \!\!\sum_{i,j,n}\!\!(\mathcal{L}_m^*[\psi_k])(x_i,y_j,t_n)\cdot f_m(u_{ij}^n)\;\Delta x\,\Delta y\,\Delta t.
\]
\medskip
\begin{itemize}
  \item Separable test functions $\psi_k = \phi_x\phi_y\phi_t$ make convolutions efficient via FFT.
  \item The resulting system $\mathbf{b}=\mathbf{G}\mathbf{w}$ is tall-and-thin: many weak-integral constraints, few candidate library terms.
  \item Test-centre locations $(x_k^0,y_k^0,t_k^0)$ are sampled uniformly inside the spatio-temporal domain.
\end{itemize}
\end{frame}

% ------ New Backup 3: Dominant-Balance Thresholding ------
\begin{frame}[noframenumbering]{Dominant-Balance Thresholding (MSTLS)}
\normalsize
Term $m$ is retained if its normalised contribution exceeds the threshold:
\[
  \Pi_m = \frac{|w_m|\;\|\mathbf{G}_{:,m}\|_2}{\|\mathbf{b}\|_2},
  \qquad
  \hat\lambda \;\le\; \Pi_m \;\le\; \tfrac{1}{\hat\lambda}.
\]
\medskip
Steps:
\begin{enumerate}
  \item OLS initialisation: $\hat{\mathbf{w}} = \mathbf{G}^\dagger\mathbf{b}$
  \item Compute $\Pi_m$ for all terms; threshold at $\hat\lambda\cdot\max_j\Pi_j$
  \item Re-fit on active support; iterate until support stabilises
  \item Repeat for 6 support-scale triplets; retain intersection
\end{enumerate}
\end{frame}

% ------ New Backup 4: Synthetic Closure with Saturation ------
\begin{frame}[noframenumbering,shrink=4]{Synthetic Closure Model: the Missing Saturation}
\normalsize
\textbf{From WSINDy (identified):}\quad $R^2_{\mathrm{wf},\rho} > 0.99$
\[
  \rho_t = -\nabla\!\cdot\mathbf{p} - c\,\nabla\!\cdot(\rho\nabla\Phi)
\]
\textbf{For stable autonomous rollout (needed):}
\[
  \rho_t = -\nabla\!\cdot\mathbf{p} - c\,\nabla\!\cdot(\rho\nabla\Phi) - \gamma\,\Delta(\rho^2)
\]
The $-\gamma\,\Delta(\rho^2)$ term is a nonlinear diffusion that arrests the Morse aggregation feedback
$\rho\uparrow\;\Rightarrow\;|\nabla\Phi|\uparrow\;\Rightarrow\;$ blowup.

\medskip
\begin{itemize}
  \item The saturation term is sub-dominant in the training window: $\Pi_\gamma < \hat\lambda$ (correctly thresholded away).
  \item Strong weak-form fit and failed rollout are therefore \emph{compatible}.
  \item Richer library or kernel-based closure could capture saturation; left for future work.
\end{itemize}
\end{frame}

\begin{frame}[shrink=12]{Shift Alignment as Symmetry Reduction}
\small
\textbf{Reference frame}
\[
  \rho_{\mathrm{ref}}
  =
  \frac{1}{T_{\mathrm{tot}}}\sum_{k=1}^{T_{\mathrm{tot}}}\rho_k .
\]

\textbf{Periodic cross-correlation}
\[
  C_k(\boldsymbol{\delta})
  =
  \left\langle
    \rho_{\mathrm{ref}},
    \mathcal{T}_{\boldsymbol{\delta}}\rho_k
  \right\rangle,
  \qquad
  \boldsymbol{\delta}_k^*
  =
  \arg\max_{\boldsymbol{\delta}\in G} C_k(\boldsymbol{\delta}),
\]
with $G=\mathbb{Z}_{n_y}\times\mathbb{Z}_{n_x}$ the discrete translation group.

\textbf{FFT evaluation}
\[
  C_k(\boldsymbol{\delta})
  =
  \mathcal{F}^{-1}\!\Bigl[
    \mathcal{F}[\rho_{\mathrm{ref}}]
    \odot
    \overline{\mathcal{F}[\rho_k]}
  \Bigr](\boldsymbol{\delta}),
\]
so all candidate shifts are scored in $O(n\log n)$ time.

\medskip
\begin{itemize}
  \item Alignment quotients out coherent bulk translation and maps the snapshot manifold to a cross-section of $\mathcal{M}/G$.
  \item Integer-pixel shifts avoid interpolation artefacts on the periodic lattice.
  \item What remains is the intrinsic shape dynamics that POD is meant to compress.
\end{itemize}
\end{frame}

\begin{frame}[shrink=10]{Restriction, Lifting, and Mass Admissibility}
\small
\textbf{Variance-stabilising transform}
\[
  \tilde\rho = \sqrt{\rho+\varepsilon},
  \qquad
  \varepsilon = 10^{-10}.
\]

\textbf{Restriction to latent coordinates}
\[
  \mathbf{y}
  =
  \mathcal{R}(\mathbf{x})
  :=
  U_m^\top(\mathbf{x}-\boldsymbol{\mu}),
  \qquad
  \mathbf{x}=\mathrm{vec}(\tilde\rho).
\]

\textbf{Lifting back to the transformed field}
\[
  \hat{\mathbf{x}}
  =
  \mathcal{L}(\mathbf{y})
  :=
  U_m\,\mathbf{y}+\boldsymbol{\mu}.
\]

\textbf{Inverse transform and mass repair}
\[
  \hat\rho = [\max(0,\hat{\mathbf{x}})]^2-\varepsilon,
  \qquad
  \mathcal{S}_{M_0}(\hat\rho)
  =
  \hat\rho\,
  \frac{M_0}{\sum_{j=1}^{d}\hat\rho_j}.
\]

\begin{itemize}
  \item The inverse spatial shift $(-\Delta y_k,-\Delta x_k)$ is applied after lifting.
  \item The fixed truncation $m=19$ is deliberately held constant across regimes for fair comparison.
  \item Mass repair is mild in practice but necessary because the $\sqrt{\cdot}$ transform is nonlinear.
 \end{itemize}
\end{frame}
